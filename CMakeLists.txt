cmake_minimum_required( VERSION 2.6 )

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/")

set(LIB mastermind)
project(${LIB})

FILE (READ "${CMAKE_CURRENT_SOURCE_DIR}/debian/changelog" DEBCHANGELOG)

string(REGEX MATCH "([0-9]+\\.[0-9]+)" DEBFULLVERSION "${DEBCHANGELOG}")
string(REGEX MATCH "([0-9]+)" MAJOR_VERSION "${DEBFULLVERSION}")
string (REGEX MATCH "([0-9]+$)" MINOR_VERSION "${DEBFULLVERSION}")

set(CMAKE_CXX_FLAGS "-Wall -pedantic -std=c++0x")

set(REQUIRED_LIBRARIES
	boost_thread
	boost_system
	boost_iostreams
	cocaine-framework-native
	cocaine-io-util
	kora-util
	msgpack
	blackhole
	)

include_directories(BEFORE
	${PROJECT_SOURCE_DIR}/include
	${PROJECT_SOURCE_DIR}/foreign
	)

aux_source_directory(src SOURCES)
set(SOURCES
	${SOURCES}
	${PROJECT_SOURCE_DIR}/src/utils.hpp
	)

add_library(${LIB} SHARED ${SOURCES})

set_target_properties(${LIB} PROPERTIES
	VERSION "${MAJOR_VERSION}.${MINOR_VERSION}"
	SOVERSION ${MAJOR_VERSION}
	)

target_link_libraries(${LIB}
	${REQUIRED_LIBRARIES})

add_subdirectory(bindings)

install(TARGETS ${LIB}
	RUNTIME DESTINATION bin COMPONENT runtime
	LIBRARY DESTINATION lib COMPONENT runtime
	ARCHIVE DESTINATION lib COMPONENT developement
	)

install(
    DIRECTORY
        include/
    DESTINATION include
    COMPONENT development)
